108  public List[] detectSpermatozoa(ImagePlus imp) {
109 
110  int nFrames = imp.getStackSize();
111  ImageStack stack = imp.getStack();
112  int options = 0; // set all PA options false
113  int measurements = MEAN + CENTROID + RECT + AREA + PERIMETER + FERET;
114  // Initialize results table
115  ResultsTable rt = new ResultsTable();
116  rt.reset();
117  int minSize = (int) (Params.minSize * Math.pow((1 / Params.micronPerPixel), 2));
118  int maxSize = (int) (Params.maxSize * Math.pow((1 / Params.micronPerPixel), 2));
119  // create storage for Spermatozoa positions
120  List[] spermatozoa = new ArrayList[nFrames];
121  // *************************************************************
122  // * Record spermatozoa positions for each frame in an ArrayList
123  // *************************************************************/
124  for (int iFrame = 1; iFrame <= nFrames; iFrame++) {
125  IJ.showProgress((double) iFrame / nFrames);
126  IJ.showStatus("Identifying spermatozoa per frame...");
127  spermatozoa[iFrame - 1] = new ArrayList();
128  rt.reset();
129  ParticleAnalyzer pa = new ParticleAnalyzer(options, measurements, rt, minSize, maxSize);
130  pa.analyze(imp, stack.getProcessor(iFrame));
131  float[] sxRes = rt.getColumn(ResultsTable.X_CENTROID);
132  float[] syRes = rt.getColumn(ResultsTable.Y_CENTROID);
133  float[] bxRes = rt.getColumn(ResultsTable.ROI_X);
134  float[] byRes = rt.getColumn(ResultsTable.ROI_Y);
135  float[] widthRes = rt.getColumn(ResultsTable.ROI_WIDTH);
136  float[] heightRes = rt.getColumn(ResultsTable.ROI_HEIGHT);
137  float[] areaRes = rt.getColumn(ResultsTable.AREA);
138  float[] perimeterRes = rt.getColumn(ResultsTable.PERIMETER);
139  float[] feretRes = rt.getColumn(ResultsTable.FERET);
140  float[] minFeretRes = rt.getColumn(ResultsTable.MIN_FERET);
141  if (sxRes == null) //Nothing detected
142  continue;//jump to next frame
143  for (int iPart = 0; iPart < sxRes.length; iPart++) {
144  Spermatozoon aSpermatozoon = new Spermatozoon();
145  aSpermatozoon.id = "***";
146  aSpermatozoon.x = sxRes[iPart];
147  aSpermatozoon.y = syRes[iPart];
148  aSpermatozoon.z = iFrame - 1;
149  aSpermatozoon.bx = bxRes[iPart];
150  aSpermatozoon.by = byRes[iPart];
151  aSpermatozoon.width = widthRes[iPart];
152  aSpermatozoon.height = heightRes[iPart];
153  aSpermatozoon.total_area = areaRes[iPart];
154  aSpermatozoon.total_perimeter = perimeterRes[iPart];
155  aSpermatozoon.total_feret = feretRes[iPart];
156  aSpermatozoon.total_minFeret = minFeretRes[iPart];
157  spermatozoa[iFrame - 1].add(aSpermatozoon);
158  }
159  }
160  return spermatozoa;
161  }

169  public SerializableList idenfityTracks(List[] spermatozoa, int nFrames) {
170 
171  // int nFrames = imp.getStackSize();
172  SerializableList theTracks = new SerializableList();
173  int trackCount = 0;
174  if(spermatozoa == null)
175  return theTracks;
176  for (int i = 0; i <= (nFrames - 1); i++) {
177  IJ.showProgress((double) i / nFrames);
178  IJ.showStatus("Calculating Tracks...");
179  if(spermatozoa[i] == null)//no spermatozoa detected in frame i
180  continue; //jump to next frame
181  for (ListIterator j = spermatozoa[i].listIterator(); j.hasNext();) {
182  Spermatozoon aSpermatozoon = (Spermatozoon) j.next();
183  if (!aSpermatozoon.inTrack) {
184  // This must be the beginning of a new track
185  List aTrack = new ArrayList();
186  trackCount++;
187  aSpermatozoon.inTrack = true;
188  aSpermatozoon.trackNr = trackCount;
189  aTrack.add(aSpermatozoon);
190  // *************************************************************
191  // search in next frames for more Spermatozoa to be added to
192  // track
193  // *************************************************************
194  boolean searchOn = true;
195  Spermatozoon oldSpermatozoon = new Spermatozoon();
196  Spermatozoon tmpSpermatozoon = new Spermatozoon();
197  oldSpermatozoon.copy(aSpermatozoon);
198  // *
199  // * For each frame
200  // *
201  for (int iF = i + 1; iF <= (nFrames - 1); iF++) {
202  boolean foundOne = false;
203  Spermatozoon newSpermatozoon = new Spermatozoon();
204  // *
205  // * For each Spermatozoon in this frame
206  // *
207  for (ListIterator jF = spermatozoa[iF].listIterator(); jF.hasNext() && searchOn;) {
208  Spermatozoon testSpermatozoon = (Spermatozoon) jF.next();
209  float distance = testSpermatozoon.distance(oldSpermatozoon);
210  // record a Spermatozoon when it is within the search
211  // radius, and when it had not yet been claimed by another
212  // track
213  if ((distance < (Params.maxDisplacement / Params.micronPerPixel)) && !testSpermatozoon.inTrack) {
214  // if we had not found a Spermatozoon before, it is easy
215  if (!foundOne) {
216  tmpSpermatozoon = testSpermatozoon;
217  testSpermatozoon.inTrack = true;
218  testSpermatozoon.trackNr = trackCount;
219  newSpermatozoon.copy(testSpermatozoon);
220  foundOne = true;
221  } else {
222  // if we had one before, we'll take this one if it is
223  // closer. In any case, flag these Spermatozoa
224  testSpermatozoon.flag = true;
225  if (distance < newSpermatozoon.distance(oldSpermatozoon)) {
226  testSpermatozoon.inTrack = true;
227  testSpermatozoon.trackNr = trackCount;
228  newSpermatozoon.copy(testSpermatozoon);
229  tmpSpermatozoon.inTrack = false;
230  tmpSpermatozoon.trackNr = 0;
231  tmpSpermatozoon = testSpermatozoon;
232  } else {
233  newSpermatozoon.flag = true;
234  }
235  }
236  } else if (distance < (Params.maxDisplacement / Params.micronPerPixel)) {
237  // this Spermatozoon is already in another track but
238  // could have been part of this one
239  // We have a number of choices here:
240  // 1. Sort out to which track this Spermatozoon really
241  // belongs (but how?)
242  // 2. Stop this track
243  // 3. Stop this track, and also delete the remainder of
244  // the other one
245  // 4. Stop this track and flag this Spermatozoon:
246  testSpermatozoon.flag = true;
247  }
248  }
249  if (foundOne)
250  aTrack.add(newSpermatozoon);
251  else
252  searchOn = false;
253  oldSpermatozoon.copy(newSpermatozoon);
254  }
255  theTracks.add(aTrack);
256  }
257  }
258  }
259  return theTracks;
260  }
